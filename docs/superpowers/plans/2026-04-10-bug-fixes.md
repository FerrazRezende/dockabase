### Task 8: Implement Auto Away Detection

**Files:**
- Create: `resources/js/composables/useIdleDetection.ts`
- Modify: `resources/js/Layouts/AuthenticatedLayout.vue`
- Modify: `resources/js/composables/useUserStatus.ts`

**Problem:** No automatic detection when user is idle/away. Status should change to "away" after inactivity and "offline" when closing the browser.

- [ ] **Step 1: Create idle detection composable**

Create `resources/js/composables/useIdleDetection.ts`:

```typescript
import { ref, onMounted, onUnmounted } from 'vue';

export type IdleStatus = 'active' | 'idle';

export interface IdleDetectionOptions {
  /** Milliseconds of inactivity before considering user idle (default: 5 minutes) */
  idleTime?: number;
  /** Check interval in milliseconds (default: 10 seconds) */
  checkInterval?: number;
  /** Callback when status changes */
  onStatusChange?: (status: IdleStatus) => void;
}

export function useIdleDetection(options: IdleDetectionOptions = {}) {
  const {
    idleTime = 5 * 60 * 1000, // 5 minutes default
    checkInterval = 10 * 1000, // Check every 10 seconds
    onStatusChange,
  } = options;

  const isIdle = ref<boolean>(false);
  const lastActivity = ref<Date>(new Date());

  let activityCheckInterval: ReturnType<typeof setInterval> | null = null;

  /**
   * Update last activity timestamp
   */
  const updateActivity = () => {
    lastActivity.value = new Date();
    if (isIdle.value) {
      isIdle.value = false;
      onStatusChange?.('active');
    }
  };

  /**
   * Check if user has been idle
   */
  const checkIdle = () => {
    const now = new Date();
    const timeSinceActivity = now.getTime() - lastActivity.value.getTime();

    if (timeSinceActivity >= idleTime && !isIdle.value) {
      isIdle.value = true;
      onStatusChange?.('idle');
    }
  };

  /**
   * Set up event listeners for user activity
   */
  const setupEventListeners = () => {
    const events = [
      'mousedown',
      'mousemove',
      'keypress',
      'scroll',
      'touchstart',
      'click',
    ];

    events.forEach((event) => {
      document.addEventListener(event, updateActivity, { passive: true });
    });
  };

  /**
   * Remove event listeners
   */
  const removeEventListeners = () => {
    const events = [
      'mousedown',
      'mousemove',
      'keypress',
      'scroll',
      'touchstart',
      'click',
    ];

    events.forEach((event) => {
      document.removeEventListener(event, updateActivity);
    });
  };

  /**
   * Start idle detection
   */
  const start = () => {
    setupEventListeners();
    activityCheckInterval = setInterval(checkIdle, checkInterval);
  };

  /**
   * Stop idle detection
   */
  const stop = () => {
    removeEventListeners();
    if (activityCheckInterval) {
      clearInterval(activityCheckInterval);
      activityCheckInterval = null;
    }
  };

  // Auto-start on mount and cleanup on unmount
  onMounted(() => {
    start();
  });

  onUnmounted(() => {
    stop();
  });

  return {
    isIdle,
    start,
    stop,
    updateActivity,
  };
}
```
